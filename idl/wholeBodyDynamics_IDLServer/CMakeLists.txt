cmake_minimum_required(VERSION 3.5)

project(wholeBodyDynamics_IDLServer)

yarp_idl_to_dir(INPUT_FILES wholeBodyDynamics_IDLServer.thrift
                    OUTPUT_DIR ${CMAKE_CURRENT_SOURCE_DIR}/autogenerated
                    SOURCES_VAR WBDIDLSERVER_SRC
                    HEADERS_VAR WBDIDLSERVER_HEADERS
                    INCLUDE_DIRS_VAR WBDIDLSERVER_INCLUDES)

add_library(wholeBodyDynamics_IDLServer STATIC ${WBDIDLSERVER_SRC} ${WBDIDLSERVER_HEADERS})
target_include_directories(wholeBodyDynamics_IDLServer PUBLIC ${WBDIDLSERVER_INCLUDES})
set_property(TARGET wholeBodyDynamics_IDLServer PROPERTY POSITION_INDEPENDENT_CODE ON)
target_link_libraries(wholeBodyDynamics_IDLServer ${YARP_LIBRARIES})

target_include_directories(wholeBodyDynamics_IDLServer PUBLIC
                            "$<BUILD_INTERFACE:${WBDIDLSERVER_INCLUDES}>"
                            "$<INSTALL_INTERFACE:${CMAKE_INSTALL_INCLUDEDIR}/wholeBodyDynamics>")

set_property(TARGET wholeBodyDynamics_IDLServer PROPERTY PUBLIC_HEADER ${WBDIDLSERVER_HEADERS})

install(TARGETS wholeBodyDynamics_IDLServer
        RUNTIME DESTINATION "${CMAKE_INSTALL_BINDIR}" COMPONENT bin
        LIBRARY DESTINATION "${CMAKE_INSTALL_LIBDIR}" COMPONENT shlib
        ARCHIVE DESTINATION "${CMAKE_INSTALL_LIBDIR}" COMPONENT lib
        PUBLIC_HEADER DESTINATION ${CMAKE_INSTALL_INCLUDEDIR}/wholeBodyDynamics)
